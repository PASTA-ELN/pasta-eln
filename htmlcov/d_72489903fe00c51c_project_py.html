126 def execute(self, command:list[Any]) -> None: 

127 """ 

128 Event if user clicks button in the center 

129 

130 Args: 

131 command (list): list of commands 

132 """ 

133 if command[0] is Command.EDIT: 

134 self.comm.formDoc.emit(self.docProj) 

135 self.comm.changeProject.emit(self.projID,'') 

136 #collect information and then change 

137 oldPath = self.comm.backend.basePath/self.docProj['-branch'][0]['path'] 

138 if oldPath.exists(): 

139 newPath = self.comm.backend.basePath/createDirName(self.docProj['-name'],'x0',0) 

140 oldPath.rename(newPath) 

141 elif command[0] is Command.DELETE: 

142 ret = QMessageBox.critical(self, 'Warning', 'Are you sure you want to delete project?',\ 

143 QMessageBox.StandardButton.Yes, QMessageBox.StandardButton.No) 

144 if ret==QMessageBox.StandardButton.Yes: 

145 #delete database and rename folder 

146 doc = self.comm.backend.db.remove(self.projID) 

147 if '-branch' in doc and len(doc['-branch'])>0 and 'path' in doc['-branch'][0]: 

148 oldPath = self.comm.backend.basePath/doc['-branch'][0]['path'] 

149 newPath = self.comm.backend.basePath/('trash_'+doc['-branch'][0]['path']) 

150 oldPath.rename(newPath) 

151 #update sidebar, show projects 

152 self.comm.changeSidebar.emit('redraw') 

153 self.comm.changeTable.emit('x0','') 

154 elif command[0] is Command.SCAN: 

155 self.comm.backend.scanProject(self.comm.progressBar, self.projID, self.docProj['-branch'][0]['path']) 

156 self.comm.changeSidebar.emit('redraw') 

157 showMessage(self, 'Information','Scanning finished') 

158 elif command[0] is Command.REDUCE_HEIGHT_HEAD: 

159 if self.infoW is not None and self.infoW.isHidden(): 

160 self.infoW.show() 

161 elif self.infoW is not None: 

162 self.infoW.hide() 

163 elif command[0] is Command.HIDE_SHOW_ITEMS: 

164 self.comm.backend.db.hideShow(self.projID) 

165 self.comm.changeSidebar.emit('') 

166 self.comm.changeTable.emit('x0','') # go back to project table 

167 elif command[0] is Command.FOLD_ALL_ITEMS and self.tree is not None: 

168 self.foldedAll = not self.foldedAll 

169 def recursiveRowIteration(index:QModelIndex) -> None: 

170 if self.tree is not None: 

171 for subRow in range(self.tree.model().rowCount(index)): 

172 subIndex = self.tree.model().index(subRow,0, index) 

173 subItem = self.tree.model().itemFromIndex(subIndex) 

174 if self.foldedAll and not subItem.text().endswith(' -'): 

175 subItem.setText(f'{subItem.text()} -') 

176 recursiveRowIteration(subIndex) 

177 return 

178 recursiveRowIteration(self.tree.model().index(-1,0)) 

179 elif command[0] is Command.HIDE: 

180 self.showAll = not self.showAll 

181 self.changeProject('','') 

182 elif command[0] is Command.ADD_CHILD: 

183 self.comm.backend.cwd = self.comm.backend.basePath/self.docProj['-branch'][0]['path'] 

184 self.comm.backend.addData('x1', {'-name':'new folder'}, [self.projID]) 

185 self.comm.changeProject.emit('','') #refresh project 

186 elif command[0] is Command.SHOW_TABLE: 

187 self.comm.changeTable.emit(command[1], self.projID) 

188 else: 

189 print("**ERROR project menu unknown:",command) 

190 return

193 def modelChanged(self, item:QStandardItem) -> None: 

194 """ 

195 After drag-drop, record changes to backend and database directly 

196 

197 Args: 

198 item (QStandardItem): item changed, new location 

199 """ 

200 #gather old information 

201 db = self.comm.backend.db 

202 stackOld = item.text().split('/')[:-1] 

203 docID = item.text().split('/')[-1] 

204 maximized = True 

205 if docID.endswith(' -'): 

206 docID = docID[:34] 

207 maximized = False 

208 doc = db.getDoc(docID) 

209 if '-branch' not in doc: 

210 return 

211 branchOldList= [i for i in doc['-branch'] if i['stack']==stackOld] 

212 if len(branchOldList)!=1: 

213 self.changeProject('','') 

214 return 

215 branchOld = branchOldList[0] 

216 childOld = branchOld['child'] 

217 branchIdx= doc['-branch'].index(branchOld) 

218 siblingsOld = db.getView('viewHierarchy/viewHierarchy', startKey=' '.join(stackOld)) 

219 siblingsOld = [i for i in siblingsOld if len(i['key'].split(' '))==len(stackOld)+1 and \ 

220 i['value'][0]>branchOld['child'] and i['value'][0]<9999] 

221 #gather new information 

222 stackNew = [] #create reversed 

223 currentItem = item 

224 while currentItem.parent() is not None: 

225 currentItem = currentItem.parent() 

226 docIDj = currentItem.text().split('/')[-1] 

227 stackNew.append(docIDj[:34] if docIDj.endswith(' -') else docIDj) 

228 stackNew = [self.projID] + stackNew[::-1] #add project id and reverse 

229 childNew = item.row() 

230 if branchOld['path'] is not None and not branchOld['path'].startswith('http'): 

231 dirNameNew= createDirName(doc['-name'],doc['-type'][0],childNew) 

232 parentDir = db.getDoc(stackNew[-1])['-branch'][0]['path'] 

233 pathNew = f'{parentDir}/{dirNameNew}' 

234 else: 

235 pathNew = branchOld['path'] 

236 siblingsNew = db.getView('viewHierarchy/viewHierarchy', startKey=' '.join(stackNew)) 

237 siblingsNew = [i for i in siblingsNew if len(i['key'].split(' '))==len(stackNew)+1 and \ 

238 i['value'][0]>=childNew and i['value'][0]<9999] 

239 logging.debug('Change project: docID -old- -new- %s|%s %i|%s %i %s', docID, str(stackOld), branchIdx, str(stackNew), childNew, pathNew) 

240 if stackOld==stackNew and childOld==childNew: #nothing changed, just redraw 

241 return 

242 # change item in question 

243 db.updateBranch(docID=docID, branch=branchIdx, stack=stackNew, path=pathNew, child=childNew) 

244 item.setText('/'.join(stackNew+[docID]) if maximized else '/'.join(stackNew+[f'{docID} -']) ) #update item.text() to new stack 

245 # change siblings 

246 for line in siblingsOld: 

247 db.updateBranch(docID=line['id'], branch=line['value'][3], child=line['value'][0]-1) 

248 for line in siblingsNew: 

249 if line['id']!=docID: 

250 db.updateBranch(docID=line['id'], branch=line['value'][3], child=line['value'][0]+1) 

251 return

254 def iterateTree(self, nodeHier:Node) -> QStandardItem: 

255 """ 

256 Recursive function to translate the hierarchical node into a tree-node 

257 

258 Args: 

259 nodeHier (Anytree.Node): anytree node 

260 

261 Returns: 

262 QtTreeWidgetItem: tree node 

263 """ 

264 #prefill docID 

265 label = '/'.join([i.id for i in nodeHier.ancestors]+[nodeHier.id]) 

266 nodeTree = QStandardItem(label) #nodeHier.name,'/'.join(nodeHier.docType),nodeHier.id]) 

267 if nodeHier.id[0]=='x': 

268 nodeTree.setFlags(Qt.ItemIsEnabled | Qt.ItemIsSelectable | Qt.ItemIsDragEnabled | Qt.ItemIsDropEnabled) # type: ignore 

269 else: 

270 nodeTree.setFlags(Qt.ItemIsEnabled | Qt.ItemIsSelectable | Qt.ItemIsDragEnabled) # type: ignore 

271 children = [] 

272 for childHier in nodeHier.children: 

273 childTree = self.iterateTree(childHier) 

274 children.append(childTree) 

275 if children: 

276 nodeTree.appendRows(children) 

277 return nodeTree
