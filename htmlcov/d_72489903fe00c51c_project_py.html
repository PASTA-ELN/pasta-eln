74 def modelChanged(self, item:QStandardItem) -> None: 

75 """ 

76 After drag-drop, record changes to backend and database directly 

77 

78 Args: 

79 item (QStandardItem): item changed, new location 

80 """ 

81 #gather old information 

82 db = self.comm.backend.db 

83 stackOld = item.text().split('/')[:-1] 

84 docID = item.text().split('/')[-1] 

85 maximized = True 

86 if docID.endswith(' -'): 

87 docID = docID[:-2] 

88 maximized = False 

89 doc = db.getDoc(docID) 

90 if '-branch' not in doc: 

91 return 

92 branchOldList= [i for i in doc['-branch'] if i['stack']==stackOld] 

93 if len(branchOldList)!=1: 

94 self.changeProject('','') 

95 return 

96 branchOld = branchOldList[0] 

97 childOld = branchOld['child'] 

98 branchIdx= doc['-branch'].index(branchOld) 

99 siblingsOld = db.getView('viewHierarchy/viewHierarchy', startKey=' '.join(stackOld)) 

100 siblingsOld = [i for i in siblingsOld if len(i['key'].split(' '))==len(stackOld)+1 and \ 

101 i['value'][0]>branchOld['child'] and i['value'][0]<9999] 

102 #gather new information 

103 stackNew = [] #create reversed 

104 currentItem = item 

105 while currentItem.parent() is not None: 

106 currentItem = currentItem.parent() 

107 docIDj = currentItem.text().split('/')[-1] 

108 stackNew.append(docIDj[:-2] if docIDj.endswith(' -') else docIDj) 

109 stackNew = [self.projID] + stackNew[::-1] #add project id and reverse 

110 childNew = item.row() 

111 if not branchOld['path'].startswith('http'): 

112 dirNameNew= createDirName(doc['-name'],doc['-type'][0],childNew) 

113 parentDir = db.getDoc(stackNew[-1])['-branch'][0]['path'] 

114 pathNew = f'{parentDir}/{dirNameNew}' 

115 else: 

116 pathNew = branchOld['path'] 

117 siblingsNew = db.getView('viewHierarchy/viewHierarchy', startKey=' '.join(stackNew)) 

118 siblingsNew = [i for i in siblingsNew if len(i['key'].split(' '))==len(stackNew)+1 and \ 

119 i['value'][0]>=childNew and i['value'][0]<9999] 

120 logging.debug('Change project: docID -old- -new- %s|%s %i|%s %i %s', docID, str(stackOld), branchIdx, str(stackNew), childNew, pathNew) 

121 if stackOld==stackNew and childOld==childNew: #nothing changed, just redraw 

122 return 

123 # change item in question 

124 db.updateBranch(docID=docID, branch=branchIdx, stack=stackNew, path=pathNew, child=childNew) 

125 item.setText('/'.join(stackNew+[docID]) if maximized else '/'.join(stackNew+[f'{docID} -']) ) #update item.text() to new stack 

126 # change siblings 

127 for line in siblingsOld: 

128 db.updateBranch(docID=line['id'], branch=line['value'][3], child=line['value'][0]-1) 

129 for line in siblingsNew: 

130 if line['id']!=docID: 

131 db.updateBranch(docID=line['id'], branch=line['value'][3], child=line['value'][0]+1) 

132 return

189 def executeAction(self) -> None: 

190 """ Any action by the buttons at the top of the page """ 

191 if hasattr(self.sender(), 'data'): #action 

192 menuName = self.sender().data() 

193 else: #button 

194 menuName = self.sender().accessibleName() 

195 if menuName=='editProject': 

196 self.comm.formDoc.emit(self.docProj) 

197 self.comm.changeProject.emit(self.projID,'') 

198 #collect information and then change 

199 oldPath = self.comm.backend.basePath/self.docProj['-branch'][0]['path'] 

200 if oldPath.exists(): 

201 newPath = self.comm.backend.basePath/createDirName(self.docProj['-name'],'x0',0) 

202 oldPath.rename(newPath) 

203 elif menuName=='deleteProject': 

204 ret = QMessageBox.critical(self, 'Warning', 'Are you sure you want to delete project?',\ 

205 QMessageBox.StandardButton.Yes, QMessageBox.StandardButton.No) 

206 if ret==QMessageBox.StandardButton.Yes: 

207 #delete database and rename folder 

208 doc = self.comm.backend.db.remove(self.projID) 

209 if '-branch' in doc and len(doc['-branch'])>0 and 'path' in doc['-branch'][0]: 

210 oldPath = self.comm.backend.basePath/doc['-branch'][0]['path'] 

211 newPath = self.comm.backend.basePath/('trash_'+doc['-branch'][0]['path']) 

212 oldPath.rename(newPath) 

213 #update sidebar, show projects 

214 self.comm.changeSidebar.emit('redraw') 

215 self.comm.changeTable.emit('x0','') 

216 elif menuName == 'scanProject': 

217 self.comm.backend.scanProject(self.comm.progressBar, self.projID, self.docProj['-branch'][0]['path']) 

218 self.comm.changeSidebar.emit('redraw') 

219 showMessage(self, 'Information','Scanning finished') 

220 elif menuName == 'projReduceWidth': 

221 if self.infoW is not None and self.infoW.isHidden(): 

222 self.infoW.show() 

223 elif self.infoW is not None: 

224 self.infoW.hide() 

225 elif menuName == 'projHideShow': 

226 self.comm.backend.db.hideShow(self.projID) 

227 self.comm.changeSidebar.emit('') 

228 self.comm.changeTable.emit('x0','') # go back to project table 

229 elif menuName == 'allFold' and self.tree is not None: 

230 self.foldedAll = not self.foldedAll

232 if self.tree is not None:

257 def iterateTree(self, nodeHier:Node) -> QStandardItem: 

258 """ 

259 Recursive function to translate the hierarchical node into a tree-node 

260 

261 Args: 

262 nodeHier (Anytree.Node): anytree node 

263 

264 Returns: 

265 QtTreeWidgetItem: tree node 

266 """ 

267 #prefill docID 

268 label = '/'.join([i.id for i in nodeHier.ancestors]+[nodeHier.id]) 

269 nodeTree = QStandardItem(label) #nodeHier.name,'/'.join(nodeHier.docType),nodeHier.id]) 

270 if nodeHier.id[0]=='x': 

271 nodeTree.setFlags(Qt.ItemIsEnabled | Qt.ItemIsSelectable | Qt.ItemIsDragEnabled | Qt.ItemIsDropEnabled) # type: ignore 

272 else: 

273 nodeTree.setFlags(Qt.ItemIsEnabled | Qt.ItemIsSelectable | Qt.ItemIsDragEnabled) # type: ignore 

274 children = [] 

275 for childHier in nodeHier.children: 

276 childTree = self.iterateTree(childHier) 

277 children.append(childTree) 

278 if children: 

279 nodeTree.appendRows(children) 

280 return nodeTree
